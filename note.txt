#Oracle DB Listener port is 1521
#Oracle APEX web port is 8080
#EPIC servers were provisioned only allow traffics through 22/80/443 ports
#iptables are enabled for port forwarding from those ports allowed to the actual servicing ports

/etc/init.d/iptables start
#forward traffics from port 80 to port 8080 (web server)
iptables -A PREROUTING -t nat -p tcp --dport 80 -j REDIRECT --to-port 8080
#forward traffics from port 443 to port 1521 (DB Listener)
iptables -A PREROUTING -t nat -p tcp --dport 443 -j REDIRECT --to-port 1521

pdb1 tnsnames.ora

Tablespace SUDS 300G
Tablespace TEMP 30G

create directory TEST_DIR '/data/sdb1/ora/db001/dumpfile'

#create SUDS account for import
create user suds identified by suds default tablespace SUDS;
grant unlimited tablespace to SUDS;
grant dba to SUDS;

impdp \
suds/????@pdb1 \
directory=TEST_DIR \
dumpfile=expdp_SUDS_TABLES_240519_01.dmp,expdp_SUDS_TABLES_240519_02.dmp,expdp_SUDS_TABLES_240519_03.dmp,expdp_SUDS_TABLES_240519_04.dmp,expdp_SUDS_TABLES_240519_05.dmp,expdp_SUDS_TABLES_240519_06.dmp

Export SUDS application from production

Import SUDS application to new production

rerun DDLs

"Scheme Type" set to No authentication in Authentication scheme

"Run on Public Pages" set to NO in Security Attributes

"Authorization Scheme" set to No applications authorization required in Security Attributes

#create SSO account for import
create user sso identified by sso default tablespace SUDS;
grant unlimited tablespace to SSO;
grant dba to SSO;

impdp \
sso/sso@pdb1 \
directory=TEST_DIR \
dumpfile=SSO_3005201901.dmp
